```typescript
// services/places/placesService.ts
import { apiClient } from '@/configs/api';
import { env } from '@/configs/env';
import type { PlaceInfo, PlaceSearchFilters, Location } from '@/types/places';
import { transformPlaceResponse } from './placesUtils';

const FIELD_MASK = [
  'places.id',
  'places.displayName',
  'places.formattedAddress',
  'places.location',
  'places.rating',
  'places.userRatingCount',
  'places.priceLevel',
  'places.photos',
  'places.currentOpeningHours',
  'places.primaryType',
  'places.googleMapsUri',
].join(',');

export const placesService = {
  /**
   * í…ìŠ¤íŠ¸ ê²€ìƒ‰ìœ¼ë¡œ ìŒì‹ì  ì¡°íšŒ
   */
  async searchPlaces(
    query: string,
    location: Location,
    filters: PlaceSearchFilters = {}
  ): Promise<{ places: PlaceInfo[]; nextPageToken?: string }> {
    const response = await apiClient.post(
      'https://places.googleapis.com/v1/places:searchText',
      {
        textQuery: query,
        locationBias: {
          circle: {
            center: {
              latitude: location.lat,
              longitude: location.lon,
            },
            radius: filters.radius || 2000,
          },
        },
        languageCode: 'ko',
        includedType: 'restaurant',
        openNow: filters.openNow,
        minRating: filters.minRating,
        priceLevels: filters.priceLevels,
        pageSize: 10,
      },
      {
        headers: {
          'X-Goog-Api-Key': env.GOOGLE_PLACES_API_KEY,
          'X-Goog-FieldMask': FIELD_MASK,
        },
      }
    );

    return {
      places: response.data.places?.map(transformPlaceResponse) || [],
      nextPageToken: response.data.nextPageToken,
    };
  },

  /**
   * ì‚¬ì§„ URL ìƒì„±
   */
  getPhotoUrl(photoName: string, maxWidth: number = 400): string {
    return `https://places.googleapis.com/v1/${photoName}/media?maxWidthPx=${maxWidth}&key=${env.GOOGLE_PLACES_API_KEY}`;
  },
};
```

### React Query Hook

```typescript
// hooks/usePlaceSearch.ts
import { useQuery, useInfiniteQuery } from '@tanstack/react-query';
import { placesService } from '@/services/places';
import type { Location, PlaceSearchFilters } from '@/types/places';

export const usePlaceSearch = (
  query: string,
  location: Location | null,
  filters: PlaceSearchFilters = {}
) => {
  return useQuery({
    queryKey: ['places', query, location?.lat, location?.lon, filters],
    queryFn: () => placesService.searchPlaces(query, location!, filters),
    enabled: !!query && !!location,
    staleTime: 5 * 60 * 1000,
    retry: 2,
  });
};

// ë¬´í•œ ìŠ¤í¬ë¡¤ìš©
export const usePlaceSearchInfinite = (
  query: string,
  location: Location | null,
  filters: PlaceSearchFilters = {}
) => {
  return useInfiniteQuery({
    queryKey: ['places-infinite', query, location?.lat, location?.lon, filters],
    queryFn: ({ pageParam }) =>
      placesService.searchPlaces(query, location!, filters, pageParam),
    getNextPageParam: (lastPage) => lastPage.nextPageToken,
    enabled: !!query && !!location,
  });
};
```

```typescript
// components/places/PlaceList.tsx
interface PlaceListProps {
  query: string;
  location: Location;
  filters: PlaceSearchFilters;
  onSelectPlace: (place: PlaceInfo) => void;
}

export const PlaceList = ({
  query,
  location,
  filters,
  onSelectPlace,
}: PlaceListProps) => {
  const { data, isLoading, error } = usePlaceSearch(query, location, filters);

  if (isLoading) return <PlaceListSkeleton />;
  if (error) return <PlaceListError error={error} />;
  if (!data?.places.length) return <PlaceListEmpty />;

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
      {data.places.map((place) => (
        <PlaceCard
          key={place.id}
          place={place}
          onClick={() => onSelectPlace(place)}
        />
      ))}
    </div>
  );
};
```
